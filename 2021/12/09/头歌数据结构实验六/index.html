void DFS(MGraph &m,int v)
{//µÝ¹éÉîËÑ 
    /*********begin***********/
    stack<int> s;
    s.push(v);
    visited[v]=1;
    cout<<m.vex[v];
    while(!s.empty()){
        int now,flag=0;
        now=s.top();
        for(int i=0;i<m.vexnum;i++){
            if(m.arcs[now][i].adj==1&&!visited[i]){
                s.push(i);
                visited[i]=1;
                cout<<m.vex[i];
                flag=1;
                break;
            }
        }
        if(flag==0){
            s.pop();
        }
    }
    /**********end************/
}

        for(int j=0;j<m->vexnum;j++){
            if(a==m->vertices[j].data){
                x=j;
            }
            if(b==m->vertices[j].data){
                y=j;
            }
        }
        ArcNode *p=new ArcNode;
        p->num=y;
        ArcNode *t=m->vertices[x].firstarc;
        if(!t){
            p->next=t;
            m->vertices[x].firstarc=p;
        }else{
            while(t->next!=NULL){
                t=t->next;
            }
            p->next=t->next;
            t->next=p;
        }

    } 
}
